#include <bits/stdc++.h>
using namespace std;

void fastIO() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
}

//-----------startofcode--------------------
bool isValid(string s) {
    stack<char> st;
    for (char c : s) {
        if (c == '(' || c == '{' || c == '[') {
            st.push(c);
        } else {
            if (st.empty()) return false;
            char top = st.top();
            st.pop();
            if ((c == ')' && top != '(') ||
                (c == '}' && top != '{') ||
                (c == ']' && top != '['))
                return false;
        }
    }
    return st.empty();
}
//-----------endofcode--------------------

int main() {
    fastIO();
    int t;
    cin >> t;
    while (t--) {
        string s;
        cin >> s;
        cout << (isValid(s) ? "true" : "false") << '\n';
    }
    return 0;
}
